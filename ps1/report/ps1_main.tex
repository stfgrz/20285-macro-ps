\section{Deterministic neoclassical growth model}

    \subsection{Sequential formulation}

        Consider the deterministic neoclassical growth model in its sequential formulation. The problem of the planner is:
        \begin{equation*}
            \max_{\{c_t, k_{t+1}\}_{t=0}^\infty} \sum_{t=0}^{\infty} \beta^t \frac{c_t^{1-\sigma}-1}{1-\sigma}
        \end{equation*}
        \begin{align*}
        \text{ s.t. }
            \begin{cases}
            c_t + k_{t+1} \leq k_t^\alpha + (1 - \delta) k_t \\
            c_t, k_{t+1} \geq 0 \\
            k_0 \text{ given}
            \end{cases}
        \end{align*}

\paragraph{1.} Use L'Hopital's rule to prove that, as $\sigma \to 1$, the within period utility function, $u(c_t) = \frac{c_t^{1-\sigma} - 1}{1-\sigma}$, goes to $\ln(c_t)$.
           
\begin{solution}
The within period utility function is 
\[\frac{c_t^{1-\sigma} -1 }{1 - \sigma}.\]
We compute the limit as follows:
\begin{align*}
   \lim_{\sigma \to 1} \frac{c_t^{1-\sigma} - 1}{1 - \sigma} & =  \lim_{\sigma \to 1} \frac{e^{(1 - \sigma)\ln(c_t)} - 1}{1 - \sigma} \\
   & =  \lim_{\sigma \to 1} \frac{\frac{\diff \left(e^{(1 - \sigma)\ln(c_t)} -1\right)}{\diff \sigma}}{\frac{\diff(1 - \sigma)}{\diff \sigma}} \\
   & = \lim_{\sigma \to 1} \ln(c_t)\,e^{(1 - \sigma)\ln(c_t)} \\
   & = \ln(c_t)
\end{align*}
where the last equality follows from the fact that the function $e^{(1-\sigma)\ln(c_t)}$ is continuous at the limit point and it is equal to 1. 
\end{solution}



\paragraph{2.} Briefly discuss which of the constraints above are binding or non-binding.

\begin{solution}
\paragraph{Claim 1.} Monotonicity of the utility function guarantees that the feasibility constraint is always satisfied with the equality at the optimum.
\begin{proof}
    
By contradiction, assume that there exists a solution $(c_t^*,k_{t+1}^*)_{t=0}^{\infty}$ such that, at time $t'$, the constraint does not bind:
\begin{equation*}
    c_{t'}^*+k_{t'+1}^*<(k_{t'}^*)^\alpha+(1-\delta)k_{t'}^*.
\end{equation*}
Then, there exists an $\varepsilon>0$ arbitrarily small such that the sequence $(\hat c_t,\hat k_{t+1})_{t=0}^\infty$ is feasible, where:
\begin{align*}
    \hat k_{t+1}=k_{t+1}^*, \qquad \forall t \geq 0 \\
    \begin{cases}
        \hat c_t=c_t^*, \qquad &\forall t \neq t' \\
        \hat c_t=c_t^*+\varepsilon, &t=t'
    \end{cases}
\end{align*}
Therefore:
\begin{equation*}
    \sum_{t=0}^\infty \beta^t \frac{(c_t^*)^{1-\sigma}-1}{1-\sigma} < \sum_{t=0}^\infty \beta^t\frac{(\hat c_t)^{1-\sigma}-1}{1-\sigma}.
\end{equation*}
contradicting the optimality of $(c_t^*,k_{t+1}^*)_{t=0}^{\infty}$. 
\end{proof}

\paragraph{Claim 2.} Non-negativity constraints never bind at optimum.

\begin{proof}
Intuitively, if we have $c_t=0$, by the Inada conditions, $\lim_{c_t \to 0} u'(c_t)= \infty$. Therefore, shifting a very small amount 
$\varepsilon>0$ to consumption at date 
$t$ yields an arbitrarily large increase in utility. For 
$\varepsilon$ sufficiently small, this first-order gain outweighs the utility loss caused by reducing consumption slightly in some other period to keep the allocation feasible. \\
Same reasoning applies for capital. Indeed, by Inada condition, $\lim_{k_t \to 0}f(k_t)=\infty$. Again, for $\eta$ sufficiently small, an increase in $k_t$ will generate a gain that will outweigh the negative effect. \\
\\
For a sketch of a formal proof see the Appendix.\end{proof}

\end{solution}

\paragraph{3.} Define the Lagrangian associated to the maximization problem above. Derive the first order necessary conditions.

\begin{solution}
Since the non negativity constraint is not binding and feasibility constraint is, we can ignore the first one and substitute the second one in the utility. Therefore the Lagrangian is 
\[
\lag = \sum_{t = 0}^{\infty} \beta^t \frac{\Big(k^\alpha_t+ (1 - \delta )k_t - k_{t+1}\Big)^{1-\sigma}-1}{1-\sigma}.
\]
The first order conditions are 
\begin{align*}
    0 = \frac{\diff \lag }{\diff k_{t+1}} = - \beta^t \Big(k^\alpha_t + (1 -\delta)k_t & - k_{t+1}\Big)^{-\sigma}\\ & +  \beta ^{t+ 1} \Big(k^\alpha_{t+1} + (1 -\delta)k_{t+1} - k_{t+2}\Big)^{-\sigma} \Big( \alpha k^{\alpha - 1}_{t + 1} + 1 - \delta\Big),
\end{align*}
for all $t \geq 0$.
We can rewrite this as 
\[ \Big(k^\alpha_t + (1 -\delta)k_t - k_{t+1}\Big)^{-\sigma} = \beta \Big(k^\alpha_{t+1} + (1 -\delta)k_{t+1} - k_{t+2}\Big)^{-\sigma} \Big( \alpha k^{\alpha - 1}_{t + 1} + 1 - \delta\Big)\]
that is
\[
u'(c_t)=\beta (\alpha k_{t+1}^{\alpha-1}+1-\delta)u'(c_{t+1}).
\]
The latter is the Euler equation, which characterizes the optimal intertemporal consumption-saving decision:
\begin{itemize}
    \item The left-hand side is the marginal utility cost of giving up one unit of consumption today.
    \item The right-hand side is the discounted marginal utility benefit generated by investing that unit instead.
\end{itemize}
\end{solution}

\paragraph{4.} Clearly list all the equations that you need to solve the model. Also state the steady state.

                \begin{solution}
We derive the transversality condition. At infinite horizon, we need to show that for any feasible path the following holds:
\[
\lim_{T\to \infty } \sum_{t=0}^\infty \beta^tu((k_t^*)^\alpha +(1-\delta)k_t^*-k_{t+1}^*) \geq \lim_{T\to \infty } \sum_{t=0}^\infty \beta^tu(k_t^\alpha +(1-\delta)k_t-k_{t+1})
\]
where $(k_{t+1}^*)_{t=0}^\infty$ denotes the optimal path of capital, while $(k_{t+1})_{t=0}^\infty$ denotes any feasible capital path. To save on notation, denote:
\begin{align*}
    u(k_t,k_{t+1})=u(k_t^\alpha +(1-\delta)k_t-k_{t+1}) 
\end{align*}
Moreover, denote $u_{1}(k_t,k_{t+1})$ and $u_{2}(k_t,k_{t+1})$ the derivatives with respect to the first and the second argument respectively. By concavity and differentiability of the utility function we have:
\[
u(k_t,k_{t+1})\leq u(k_t^*,k_{t+1}^*)+u_{1}(k_t^*,k_{t+1}^*)(k_t-k_t^*)+u_{2}(k_t^*,k_{t+1}^*)(k_{t+1}-k_{t+1}^*).
\]
Multiplying by $\beta^t$ and summing up the first $T$ terms one gets:
\[
\sum_{t=0}^T \beta^tu(k_t,k_{t+1})\leq \sum_{t=0}^T \beta^tu(k^*_t,k_{t+1}^*)+D_T
\]
where $D_T=\sum_{t=0}^T\beta^tu_{1}(k_{t}^*,k_{t+1}^*)(k_t-k_{t}^*)+\sum_{t=0}^T\beta^tu_2(k_{t}^*,k_{t+1}^*)(k_{t+1}-k_{t+1}^*)$. \\
We want to show that $\lim_{T\to \infty}D_T\leq 0$. Let's rewrite $D_T$ as follows:
\begin{align*}
    D_T&=u_{1}(k_0^*,k_1^*)(k_0-k_0^*)+u_{2}(k_0^*,k_1^*)(k_1-k_1^*) \\
    &+\beta u_{1}(k_1^*,k_2^*)(k_1-k_1^*)+u_{2}(k_1^*,k_2^*)(k_2-k_2^*)\\
    & + \dots + \beta^Tu_{1}(k_T^*,k_{T+1}^*)(k_{T}-k_{T}^*)+\beta^Tu_{2}(k_T^*,k_{T+1}^*)(k_{T+1}-k_{T+1}^*)
\end{align*}
Notice that we can rearrange the terms in $D_T$ and obtain:
\begin{align*}
D_T&=u_1(k_0^*,k_1^*)(k_0-k_0^*)\\&+\sum_{t=0}^T\beta^t[u_2(k_t^*,k_{t+1}^*)+\beta u_1(k_{t+1}^*,k_{t+2}^*)](k_{t+1}-k_{t+1}^*)\\&-\beta^{T+1} u_1(k_{T+1}^*,k_{T+2}^*)(k_{T+1}-k_{T+1}^*)
\end{align*}
Recall the first order condition:
\[
u_{2}(k_t^*,k_{t+1}^*)+\beta u_{1}(k_{t+1}^*,k_{t+2}^*)=0
\]
Then, we can simplify the expression for $D_t$:
\begin{align*}
    D_T=u_{1}(k_0^*,k_1^*)(k_0-k_0^*)-\beta^{T+1} u_{1}(k_{T+1}^*,k_{T+2}^*)(k_{T+1}-k_{T+1}^*).
\end{align*}
Since $k_0=k_0^*$, we have:
\begin{equation*}
    \lim_{T\to \infty}D_T=\lim_{T \to \infty }-\beta^{T+1} u_{1}(k_{T+1}^*,k_{T+2}^*)(k_{T+1}-k_{T+1}^*) \leq 0
\end{equation*}
We can write it extensively:
\begin{align*}
    \lim_{T \to \infty } -\beta^{T+1} (\alpha(k_{T+1}^*)^{\alpha-1}+1-\delta)u'((k_{T+1}^*)^{\alpha}+(1-\delta)k_{T+1}^*-k_{T+2}^*)(k_{T+1}-k_{T+1}^*)\leq 0
\end{align*}
that is
\begin{align*}
    \lim_{T \to \infty } \beta^{T+1} f'(k_{T+1})u'(c_{T+1}^*) (k_{T+1}-k_{T+1}^*)\geq 0
\end{align*}
Now notice that, since the marginal utility is always positive and $k_{T+1}\geq 0$:
\begin{align*}
    \lim_{T \to \infty } \beta^Tf'(k_{T+1})u'(c_{T+1}^*) (k_{T+1}-k_{T+1}^*)  \geq -\lim_{T \to \infty } \beta^Tf'(k_{T+1})u'(c_{T+1}^*)k_{T+1}^* \geq 0
\end{align*}
which implies:
\[
\lim_{T \to \infty } \beta^Tf'(k_{T+1})u'(c_{T+1}^*)k_{T+1}^*=0
\]
 \\
 The \textbf{conditions for the optimum} are, for all $t\geq 0$, the Euler equations and the binding feasibility constraints: 
 \begin{align*}
     u'(c_t)&=\beta (\alpha k_{t+1}^{\alpha-1}+1-\delta)u'(c_{t+1}), \qquad  \tag{EE}\\
     c_t + k_{t+1} &= k_t^\alpha + (1 - \delta) k_t \qquad \tag{FC}
 \end{align*}
 and the transversality condition:
 \begin{align*}
          \lim_{T \to \infty } \beta^T&f'(k_{T+1})u'(c_{T+1}^*)k_{T+1}^*=0 \qquad \tag{TVC}
 \end{align*}
 Lastly, to find the \textbf{steady state}, we impose $k_t=k_{t+1}=K$ and $c_t=c_{t+1}=C$:
 \begin{align*}
     1&=\beta (\alpha K^{\alpha-1}+1-\delta) \\
     C&=K^\alpha-\delta K.
 \end{align*}
 Rearranging the expressions:
 \begin{align*}
     K&= \left [ \frac{1}{\alpha}\left(\frac{1}{\beta}-(1-\delta)\right)\right]^{\frac{1}{\alpha-1}}=\left( \frac{1-\beta(1-\delta)}{\alpha \beta}\right)^{\frac{1}{\alpha-1}} \\
     C&=\left( \frac{1-\beta(1-\delta)}{\alpha \beta}\right)^{\frac{\alpha}{\alpha-1}}-\delta \left( \frac{1-\beta(1-\delta)}{\alpha \beta}\right)^{\frac{1}{\alpha-1}}
 \end{align*}
                \end{solution}

\paragraph{5. }Approximate the system of non-linear difference equations around the model's deterministic steady state by a first-order Taylor expansion, i.e. log-linearize it. Give a brief intuition why this is helpful to solve the model.

                \begin{solution}
First, we start with the \textbf{resource constraint}:
\begin{align*}
    c_t+k_{t+1}=k_t^\alpha+(1-\delta)k_t.
\end{align*}
Then, we take a log-linear approximation around the steady state:
\begin{align}
    C(1+\hat c_t)+K(1+\hat k_{t+1})=K^\alpha(1+\alpha\hat k_{t+1})+(1-\delta)K(1+\hat k_t). \label{suca}
\end{align}
We know that, at the steady state, the resource constraint becomes:
\[
C+K=K^\alpha+(1-\delta)K.
\]
Hence, we can simplify equation \eqref{suca}:
\[
C\hat c_t+K\hat k_{t+1}=K^\alpha \alpha \hat k_t+(1-\delta)K\hat k_t
\]
Then, divide everything by $K$:
\begin{equation}
    \frac{C}{K}\hat c_t+\hat k_{t+1}=(\alpha K^{\alpha-1}+1-\delta)\hat k_t. \label{sucaduevolte}
\end{equation}
Now notice that the Euler Equation in steady state guarantees
\[
\alpha K^{\alpha-1}+1-\delta =\frac{1}{\beta}.
\]
Therefore, \eqref{sucaduevolte} becomes
\[
\frac{C}{K}\hat c_t+\hat k_{t+1}=\frac{1}{\beta}\hat k_t
\]
Let's move to the \textbf{Euler equation}:
\[
1=\beta(\alpha k_{t+1}^{\alpha-1}+1-\delta)\left( \frac{c_{t+1}}{c_t}\right)^{-\sigma}.
\]
Now, we rewrite all variables as follows:
\[
1=\beta(\alpha K^{\alpha-1}e^{(\alpha-1)\hat k_{t+1}}+1-\delta)\left( \frac{C^{-\sigma}e^{-\sigma \hat c_{t+1}}}{C^{-\sigma}e^{-\sigma \hat c_{t}}}\right).
\]
Then, factor together all the exponential terms:
\begin{align*}
    1= \beta \alpha K^{\alpha-1}e^{(\alpha-1)\hat k_{t+1}-\sigma(\hat c_{t+1}-\hat c_t)}+\beta(1-\delta)e^{-\sigma(\hat c_{t+1}-\hat c_t)}.
\end{align*}
Then, take a first order Taylor approximation around the steady state:
\begin{align*}
    1 \simeq \beta \alpha K^{\alpha-1}(1+(\alpha-1)\hat k_{t+1}-\sigma(\hat c_{t+1}-\hat c_t))+\beta (1-\delta)(1-\sigma(\hat c_{t+1}-\hat c_t)),
\end{align*}
and
\begin{align*}
    1\simeq \beta[\alpha K^{\alpha-1}+(1-\delta)]+\beta \alpha K^{\alpha-1}((\alpha-1)\hat k_{t+1}-\sigma(\hat c_{t+1}-\hat c_t))-\beta\sigma(1-\delta)(\hat c_{t+1}-\hat c_t).
\end{align*}
By the Euler equation in steady state, we know that $\beta[\alpha K^{\alpha-1}+(1-\delta)]=1$; hence:
\begin{align*}
    1&\simeq 1+\beta \alpha K^{\alpha-1}((\alpha-1)\hat k_{t+1}-\sigma(\hat c_{t+1}-\hat c_t))-\beta\sigma(1-\delta)(\hat c_{t+1}-\hat c_t) \\
    0 &\simeq \beta \alpha (\alpha-1)K^{\alpha-1}\hat k_{t+1}-\sigma \beta[\alpha K^{\alpha-1}+(1-\delta)](\hat c_{t+1}-\hat c_t)\\
    0 &\simeq \beta(\alpha-1)\left(\frac{1}{\beta}-(1-\delta)\right) \hat k_{t+1}-\sigma(\hat c_{t+1}-\hat c_t).
\end{align*}
Therefore, the log-linearized version of Euler equation is:
\begin{equation*}
    (\alpha-1)[1-\beta(1-\delta)]\hat k_{t+1} \simeq \sigma(\hat c_{t+1}-\hat c_t)
\end{equation*}
Log-linearization transforms a system of nonlinear equations into linear ones, in terms of log-devations of the variables from their steady-state values. This approximation allows for a clearer interpretation of the dynamics of the model and, more importantly, for an (approximate) analytical solution.\end{solution}

\paragraph{6.} After deriving the optimal sequence for consumption and capital in period $t = 0$, denoted $\{c_t^*, k_{t+1}^*\}_{t=0}^\infty$, let's assume our planner reassesses the situation in period 1, one period later. 

\begin{enumerate}
    \item [a.] Does she need to redo all the algebra, i.e., solve the entire maximization problem starting from $t = 1$, or can she simply follow the plan set in the previous period? 
    
\begin{solution}
     By the Bellman principle of optimality, if a path $(c_t^*,k_{t+1}^*)_{t=0}^\infty$ solves the problem at $t=0$, then it is time-consistent, that is,   $(c_t^*,k_{t+1}^*)_{t=1}^\infty$ will solve the problem that starts at $t=1$.                   
\end{solution}

                        
    \item [b.] What happens if, unexpectedly, there's a small change to the capital stock at the start of period 1 in form of a so-called MIT shock, so that the available capital $k_1$ differs from what was planned in the optimal sequence, i.e., $k_1 \neq k_1^*$?

                        \begin{solution}
At time $t=1$, agents observe the shock, and they reoptimize, finding a new optimal sequence. Hence, the Bellman principle of optimal does not hold anymore. Intuitively, the resource constraint at time $t=1$ changed: it is now with $k_1 \neq k_1^*$. From Claim 1, we know that it has to be binding; therefore, the whole optimal sequence will change. 
                        \end{solution}
\end{enumerate}

\paragraph{7. } Now, formulate the problem in recursive form: Clearly identify the state variable, the control variables, and write down the Bellman equation.

                \begin{solution}
The state variable is capital today $k_t$, the control variables are consumption today $c_t$ and capital tomorrow $k_{t+1}$. The Bellman equation is 
\begin{align*}
V(k_t)=&\max_{\{c_t,k_{t+1}\}}\left(\frac{c_t^{1-\sigma}}{1-\sigma}+\beta V(k_{t+1})\right) \\&\text{ s.t. } \begin{cases} c_t \leq k_t^\alpha+(1-\delta)k_t-k_{t+1} \\ c_t,k_{t+1} \geq 0
\end{cases}
\end{align*}
                    
                    
                \end{solution}
            
\paragraph{8.} Derive the first order necessary conditions for the problem from the Bellman equation you derived in the previous point.  

                \begin{solution}

                    By Claim 1, the resource constraint is binding so we can substitute it. By Claim 2, instead, non-negativity constraints do not bind. Then the Bellman equation is 
\begin{gather*}
V(k_t)=\max_{\{k_{t+1}\}}\left(\frac{(k_t^\alpha+(1-\delta)k_t-k_{t+1})^{1-\sigma}}{1-\sigma}+\beta V(k_{t+1})\right) 
\end{gather*}
and first order condition is 
\begin{equation*}
    -(k_t^\alpha+(1-\delta)k_t-k_{t+1})^{-\sigma}+\beta V'(k_{t+1}) =0.
\end{equation*}
                    
                \end{solution}

\paragraph{9.} Exploit the envelope condition(s) to express the first order conditions above in terms of the variables of the model. Comment on the Euler equation you obtain and compare it to the one you obtained in \textbf{Part 1}.  


                \begin{solution}
The value function $V(\cdot)$ pins down the policy function $k_{t+1}=g(k_t)$. 

\noindent By the envelope theorem, 
\begin{equation*}
    V'(k_t)=(k_t^\alpha+(1-\delta)k_t-g(k_t))^{-\sigma}(\alpha k_t^{\alpha-1}+(1-\delta)).
\end{equation*}

\noindent By the first order condition, 
\begin{equation*}
    \beta V'(k_{t+1})=(k_t^\alpha+(1-\delta)k_t-k_{t+1})^{-\sigma}.
\end{equation*}

\noindent Hence
\begin{gather*}
    (k_t^\alpha+(1-\delta)k_t-k_{t+1})^{-\sigma}=\beta(k_{t+1}^\alpha+(1-\delta)k_{t+1}-k_{t+2})^{-\sigma}(\alpha k_{t+1}^{\alpha-1}+(1-\delta)).
\end{gather*}

The Euler equations we obtained in \textbf{Part 1} and \textbf{Part 2} are identical because the solutions to the recursive and sequential formulations of the planner problem are identical, by the optimality principle. 

                    
                    
                \end{solution}

\paragraph{10.} Consider the nested case where (i) $\sigma = 1$, (ii) $\delta = 1$. Guess the following policy functions for the optimal decisions of $c_t$ and $k_{t+1}$ in terms of the state $k_t$:
                \begin{align*}
                    c_t & = f(k_t) = A k_t^\alpha\\
                    k_{t+1} & = g(k_t) = B k_t^\alpha.
                \end{align*}
            Find the values for the undetermined coefficients $A$ and $B$.  

\begin{solution} Given the parameters, the functional Euler equation is 
\begin{gather*}
    \frac{1}{k_t^\alpha-k_{t+1}}=\frac{\beta\alpha k_{t+1}^{\alpha-1}}{k_{t+1}^\alpha-k_{t+2}}
\end{gather*}
that we can rewrite as the system of equations
\begin{gather}
    \frac{1}{c_t}=\frac{\beta\alpha k_{t+1}^{\alpha-1}}{c_{t+1}} \\ c_t+k_{t+1}=k_t^\alpha
\end{gather}
and then plug in the guesses 
\begin{gather}
    \frac{1}{Ak_t^\alpha}=\frac{\beta\alpha (Bk_t^\alpha)^{\alpha-1}}{Ak_{t+1}^\alpha}=\frac{\beta\alpha (Bk_t^\alpha)^{\alpha-1}}{A(Bk_t^\alpha)^\alpha} \\ Ak_t^\alpha+Bk_t^\alpha=k_t^\alpha.
\end{gather}
First, 
\begin{gather*}
     \frac{1}{Ak_t^\alpha}=\frac{\beta\alpha (Bk_t^\alpha)^{\alpha-1}}{A(Bk_t^\alpha)^\alpha}\iff \frac{1}{k_t^\alpha} = \frac{\beta\alpha}{Bk_t^\alpha} \iff 1 = \frac{\beta\alpha}{B} \iff B = \alpha\beta.
\end{gather*}
Then, 
\begin{gather*}
    Ak_t^\alpha+\alpha\beta k_t^\alpha=k_t^\alpha \iff A = 1-\alpha \beta.
\end{gather*}

                    
                \end{solution}

\paragraph{11.} After deriving the policy functions for consumption and capital, let's again assume our planner reassesses the situation in period 1, one period later. 

                \begin{enumerate}
                    \item[a.] Should she still follow the consumption-saving choices prescribed by the policy functions, given capital stock $k_1 = g(k_0) = B k_0^\alpha$? 

                        \begin{solution}
                        In the absence of shocks, then Bellman principle of optimality holds. Hence, if the optimal solution for the problem at time $t=0$ was $(c_t^*,k_{t+1}^*)_{t=0}^\infty$, found applying iteratively the policy function to the initial condition $k_0$, then $(c^*_t,k_{t+1}^*)_{t=1}^\infty$ is optimal for the problem at time $t=1$.\end{solution}
                    
                    \item[b.] What if, unexpectedly, there's a change to the capital stock in period 1, so that the available capital $k_1$ differs from what she saved according to the policy functions, i.e., $k_1 \neq B k_0^\alpha$.

                        \begin{solution}
                        In the presence of a shock, Bellman principle of optimality does not hold anymore: the agent will change the optimal sequence. However, the policy functions prescribe the optimal choice for consumption and capital given every possible value of the states variables, in our case the installed capital $k_t$.      \end{solution}
                        
                \end{enumerate}
                


\section{Modified neoclassical growth model}

    Consider the following modification of the neoclassical growth model, where we assume that investments made in period $t$ will become productive only in period $t + 2$. That is, the budget constraint is given by:
    \begin{equation}
        c_t + k_{t+1} = f(k_{t-1}) + (1 - \delta) k_t
    \end{equation}
    
    \noindent The maximization problem is hence given by:
    \begin{equation}
        \max_{\{c_t, k_{t+1}\}_{t=0}^\infty} \sum_{t=0}^{\infty} \beta^t \frac{c_t^{1-\sigma}}{1-\sigma}
    \end{equation}
    
    \noindent s.t.
    \begin{align}
        \begin{cases}
        c_t + k_{t+1} = f(k_{t-1}) + (1 - \delta) k_t \\
        c_t, k_{t+1} \geq 0 \\
        k_{-1}, k_0 \text{ given}
        \end{cases}
    \end{align}

    \noindent where production only uses capital, according to $f(k_{t-1}) = k_{t-1}^\alpha$.


\paragraph{1.} Formulate the problem in recursive form: identify the state and the control variables of the problem, and write down the Bellman equation.  
\begin{solution}
State variables are:
\begin{itemize}
    \item installed capital today $k$;
    \item productive capital today $x$.
\end{itemize}
Control variables are:
\begin{itemize}
    \item consumption today $c$;
    \item installed capital tomorrow $k'$;
    \item productive capital two-periods ahead $x''$.
\end{itemize}
Law of motions are:
\begin{itemize}
    \item $c+k'=f(x)+(1-\delta)k$;
    \item $k=x'$.
\end{itemize}
Then, the recursive formulation of the problem is:
\begin{align*}
    V(k,x)=&\max_{c,k',x''}\frac{c^{1-\sigma}}{1-\sigma}+\beta V(k',x')\\
    &\text{s.t.} \begin{cases}
        c+k'=f(x)+(1-\delta)k\\
        k=x' \\
        c,k' \geq 0
    \end{cases}
\end{align*}
Then, we can write it more compactly as follows:
\begin{align*}
    V(k,k_{-1})&=\max_{c,k'} \frac{c^{1-\sigma}}{1-\sigma}+\beta V(k',k) \\
    &\text{s.t.} \begin{cases}
        c+k'=f(k_{-1})+(1-\delta)k \\
        c,k' \geq 0
    \end{cases}
\end{align*}
\end{solution}


\paragraph{2.} Derive the first order necessary conditions for the problem from the Bellman equation you derived in the previous point.  
\begin{solution}
    As we said in Exercise 1, negativity constraints constraints never bind at optimum (see Appendix). First order conditions are:
    \begin{align}
        c: \qquad &c^{-\sigma}-\lambda=0 \label{focC}\\
        k': \qquad &\beta \frac{\partial V(k',k)}{\partial k'}-\lambda=0 \label{focK}
    \end{align}
\end{solution}


\paragraph{3.} Exploit the envelope condition(s) to express the first order conditions above in terms of the variables of the model. Comment on the Euler equation you obtain.  
\begin{solution}
    We derive the envelope conditions:
    \begin{align*}
        & \frac{\partial V(k,k_{-1})}{\partial k}=V_1(k,k_{-1})=\beta V_2(k',k)+\lambda(1-\delta) \\
        & \frac{\partial V(k,k_{-1})}{\partial k_{-1}}=V_2(k,k_{-1})=\lambda f'(k_{-1})
    \end{align*}

Then, we can use them to rearrange \eqref{focK}:
\begin{align*}
    \beta V_1(k',k)-\lambda = \beta(\beta \lambda''f'(k')+\lambda'(1-\delta))-\lambda =0.
\end{align*}
Lastly, using \eqref{focC}:
\[
c^{-\sigma}=\beta^2 \alpha (k')^{\alpha-1}(c'')^{-\sigma}+\beta (1-\delta)(c')^{-\sigma}.
\]
This Euler equation summarizes the household’s optimal intertemporal consumption--saving trade-off. 

\begin{itemize}
    \item The left-hand side, $u'(c)$, is the marginal utility cost of giving up one unit of consumption today.
    
    \item The right-hand side is the discounted marginal utility benefit generated by investing that unit instead. Because capital becomes productive only with a two-period delay, the payoff arrives through two channels:
    \begin{enumerate}
        \item Next period (installed but not yet productive): the additional capital carried into the next period yields $(1-\delta)$ units of capital after depreciation, valued at marginal utility $u'(c')$ and discounted by $\beta$, giving the term $\beta(1-\delta)u'(c')$.
        
        \item Two periods ahead (productive): once that capital becomes productive after two periods, it raises output by $f'(k')$, valued at marginal utility $u'(c'')$ and discounted by $\beta^2$, giving the term $\beta^2 f'(k')u'(c'')$.
    \end{enumerate}
\end{itemize}
\end{solution}

Simultaneously, firms halt hiring, causing vacancies to plummet. This dual effect causes the V/U ratio to collapse to near-zero levels.

                        \item \textbf{Expansions and the Post-2020 Anomaly}: During recoveries, unemployment gradually falls while vacancies rise, steadily pushing the V/U ratio up. The post-2020 period is particularly notable; the V/U ratio surged to an unprecedented peak near 2.0, meaning there were roughly two job openings for every unemployed person.                     \end{itemize}

Even if a tight labour market successfully drives up nominal wages, an unexpected or rapid increase in the general price level can completely offset those gains, causing real wage growth to stagnate or turn negative. Thus, the real wage series reflects a constant tug-of-war between nominal wage bargaining and price inflation, making it far more volatile than unemployment alone.

Because employers are notoriously reluctant to cut \textit{nominal} wages (due to morale, fairness, and productivity concerns), and because recessions typically suppress inflation, the real wages of workers who manage to retain their jobs often remain sticky or even rise slightly during contractions.
                    \end{itemize}
